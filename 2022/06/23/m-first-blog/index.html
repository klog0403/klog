var gulp = require（’gulp’）；

gulp.task（“任务名”， function（）{ //任务名设置js 函数里面return gulp.src（’目标源文件’）
//配置任务的操作
return gulp.src（’’）
.pipe（concat（’合并好的文件名.js’））//临时合并文件
.pipe（gulp.dest(‘dist/js/’)）//输出文件到本地
.pipe(uglify()) //压缩文件
.pipe(rename({suffix: ‘.min’})) //重命名
.pipe(gulp.dest(‘dist/js’))

}）

//注册默认任务
gulp.task（“default”，[]）
